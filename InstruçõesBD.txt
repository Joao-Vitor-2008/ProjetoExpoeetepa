CREATE DATABASE ProjetoExpoeetepaBD CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

USE ProjetoExpoeetepaBD;


USE ProjetoExpoeetepaBD;

CREATE TABLE estacoes (
    nome VARCHAR(32),
    temperaturaAr FLOAT,
    umidadeAr FLOAT,
    pressaoAr FLOAT,
    data_hora DATETIME,
    indice_uv INT,
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE irrigadores (
    plantio VARCHAR(32) NOT NULL UNIQUE,
    umidadeSolo FLOAT,
    acaoAtual ENUM('ligar','desligar'),
    tempoRestante INT,
    cicloDias INT,
    limiarUmidade FLOAT,
    KEY fk_estacao (data_hora),
    CONSTRAINT fk_estacao FOREIGN KEY (data_hora) REFERENCES estacoes(data_hora) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;



CREATE USER 'admin'@'localhost' IDENTIFIED BY 'senha';
GRANT SELECT, INSERT, UPDATE, DELETE ON ProjetoExpoeetepaBD.* TO 'admin'@'localhost';
FLUSH PRIVILEGES;
