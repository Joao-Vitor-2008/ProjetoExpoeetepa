CREATE DATABASE ProjetoExpoeetepaBD CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

USE ProjetoExpoeetepaBD;

CREATE TABLE estacoes (
    id INT NOT NULL AUTO_INCREMENT,
    nome VARCHAR(32),
    temperaturaAr FLOAT,
    umidadeAr FLOAT,
    pressaoAr FLOAT,
    data_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE irrigadores (
    id INT(3) NOT NULL AUTO_INCREMENT,
    id_estacao INT,
    plantio VARCHAR(32),
    umidadeSolo FLOAT,
    acaoAtual ENUM('ligar','desligar'),
    tempoRestante INT,
    cicloDias INT,
    limiarUmidade FLOAT,
    PRIMARY KEY (id),
    KEY fk_estacao (id_estacao),
    CONSTRAINT fk_estacao FOREIGN KEY (id_estacao) REFERENCES estacoes(id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;



CREATE USER 'admin'@'localhost' IDENTIFIED BY 'senha';
GRANT SELECT, INSERT, UPDATE, DELETE ON ProjetoExpoeetepaBD.* TO 'admin'@'localhost';
FLUSH PRIVILEGES;
